```
└── 📁${domain.org}  // 组织机构或者公司名倒过来,比如: org.company
    └── 📁${module}  // 项目或者模块的名称
        └── 📁annotations    // 自定义注解
        └── 📁aop    // 切面
        └── 📁configs    // 配置类
        └── 📁context    // 上下文
        └── 📁controller   // 控制器
        └── 📁cron   // 定时任务
        └── 📁dto    // 数据传输对象
        └── 📁listener   // 监听器
        └── 📁payload    // 请求载荷
        ├── 📁service  // 服务
            ├── 📁impl // 服务实现
```
